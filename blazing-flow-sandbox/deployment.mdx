## Docker Compose (Recommended)

### docker-compose.yml

```yaml
version: '3.8'

services:
  # Redis - shared state
  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    command: redis-server --appendonly yes

  # API - HTTP/WebSocket endpoints
  api:
    image: blazing/api:latest
    ports:
      - "8000:8000"
    environment:
      - REDIS_URL=redis://redis:6379
      - API_TOKEN=${API_TOKEN}
    depends_on:
      - redis
    restart: unless-stopped

  # Foreman - coordinator (does NOT run user code)
  foreman:
    image: blazing/foreman:latest
    environment:
      - REDIS_URL=redis://redis:6379
      - PILOT_LIGHT_MIN_P=2
      - PILOT_LIGHT_MIN_A=2
      - PILOT_LIGHT_ASYNC_SLOTS=5
    depends_on:
      - redis
    restart: unless-stopped

  # Regular executor - TRUSTED workers only
  executor:
    image: blazing/executor:latest
    environment:
      - REDIS_URL=redis://redis:6379
      - WORKER_TYPES=BLOCKING,NON_BLOCKING
      - DATABASE_URL=${DATABASE_URL}  # Your DB credentials
      - API_KEY=${API_KEY}            # Your API keys
    depends_on:
      - redis
      - foreman
    deploy:
      replicas: 3  # Scale as needed
    restart: unless-stopped

  # Pyodide executor - SANDBOXED workers only
  pyodide-executor:
    image: blazing/pyodide-executor:latest
    environment:
      - REDIS_URL=redis://redis:6379
      - WORKER_TYPES=BLOCKING_SANDBOXED,NON_BLOCKING_SANDBOXED
      # NO credentials - sandboxed workers don't need them
    depends_on:
      - redis
      - foreman
    deploy:
      replicas: 2  # Fewer sandboxed workers typically needed
    restart: unless-stopped

volumes:
  redis-data:
```

### Start Services

```bash
# Create .env file
cat > .env <<EOF
API_TOKEN=your-secret-token
DATABASE_URL=postgresql://user:pass@db:5432/mydb
API_KEY=your-api-key
EOF

# Start all services
docker-compose up -d

# Check status
docker-compose ps

# View logs
docker-compose logs -f foreman
docker-compose logs -f executor
docker-compose logs -f pyodide-executor
```

## Kubernetes

### Namespace

```yaml
# namespace.yaml
apiVersion: v1
kind: Namespace
metadata:
  name: blazing
```

### Redis Deployment

```yaml
# redis-deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: redis
  namespace: blazing
spec:
  replicas: 1
  selector:
    matchLabels:
      app: redis
  template:
    metadata:
      labels:
        app: redis
    spec:
      containers:
      - name: redis
        image: redis:7-alpine
        ports:
        - containerPort: 6379
        volumeMounts:
        - name: redis-data
          mountPath: /data
        command: ["redis-server", "--appendonly", "yes"]
      volumes:
      - name: redis-data
        persistentVolumeClaim:
          claimName: redis-pvc
---
apiVersion: v1
kind: Service
metadata:
  name: redis
  namespace: blazing
spec:
  selector:
    app: redis
  ports:
  - port: 6379
    targetPort: 6379
```

### API Deployment

```yaml
# api-deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: blazing-api
  namespace: blazing
spec:
  replicas: 3
  selector:
    matchLabels:
      app: blazing-api
  template:
    metadata:
      labels:
        app: blazing-api
    spec:
      containers:
      - name: api
        image: blazing/api:latest
        ports:
        - containerPort: 8000
        env:
        - name: REDIS_URL
          value: "redis://redis:6379"
        - name: API_TOKEN
          valueFrom:
            secretKeyRef:
              name: blazing-secrets
              key: api-token
        resources:
          requests:
            memory: "256Mi"
            cpu: "250m"
          limits:
            memory: "512Mi"
            cpu: "500m"
        livenessProbe:
          httpGet:
            path: /v1/health
            port: 8000
          initialDelaySeconds: 10
          periodSeconds: 30
---
apiVersion: v1
kind: Service
metadata:
  name: blazing-api
  namespace: blazing
spec:
  type: LoadBalancer
  selector:
    app: blazing-api
  ports:
  - port: 80
    targetPort: 8000
```

### Deploy to Kubernetes

```bash
# Create namespace
kubectl apply -f namespace.yaml

# Create secrets
kubectl create secret generic blazing-secrets \
  --namespace=blazing \
  --from-literal=api-token=your-secret-token \
  --from-literal=database-url=postgresql://... \
  --from-literal=api-key=your-api-key

kubectl apply -f redis-deployment.yaml

kubectl logs -f deployment/blazing-foreman -n blazing
kubectl logs -f deployment/blazing-executor -n blazing
kubectl logs -f deployment/blazing-pyodide-executor -n blazing
```

## Scaling

### Horizontal Scaling

**Trusted Executors:**
```bash
# Docker Compose
docker-compose up -d --scale executor=10

# Kubernetes
kubectl scale deployment blazing-executor --replicas=10 -n blazing
```

**Sandboxed Executors:**
```bash
# Docker Compose
docker-compose up -d --scale pyodide-executor=5

# Kubernetes
kubectl scale deployment blazing-pyodide-executor --replicas=5 -n blazing
```

### Auto-Scaling (Kubernetes)

```yaml
# hpa-executor.yaml
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: blazing-executor-hpa
  namespace: blazing
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: blazing-executor
  minReplicas: 3
  maxReplicas: 20
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 70
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 80
```

## Environment Variables

### Required

| Variable | Description | Example |
|----------|-------------|---------|
| `REDIS_URL` | Redis connection URL | `redis://redis:6379` |
| `API_TOKEN` | API authentication token | `your-secret-token` |

### Optional

| Variable | Description | Default |
|----------|-------------|---------|
| `WORKER_TYPES` | Comma-separated worker types | `NON_BLOCKING` |
| `PILOT_LIGHT_MIN_P` | Minimum blocking workers | `1` |
| `PILOT_LIGHT_MIN_A` | Minimum async workers | `1` |
| `PILOT_LIGHT_ASYNC_SLOTS` | Minimum async slots | `3` |
| `LOG_LEVEL` | Logging level | `INFO` |

### Trusted Workers Only

| Variable | Description | Example |
|----------|-------------|---------|
| `DATABASE_URL` | Database connection | `postgresql://...` |
| `API_KEY` | External API key | `your-api-key` |
| `AWS_ACCESS_KEY_ID` | AWS credentials | `AKIA...` |
| `AWS_SECRET_ACCESS_KEY` | AWS secret | `...` |

**⚠️ NEVER set credentials on sandboxed workers!**

## Security Checklist

### Trusted Workers

- [ ] Use secrets management (Kubernetes Secrets, AWS Secrets Manager)
- [ ] Rotate credentials regularly
- [ ] Use least-privilege database users
- [ ] Enable SSL/TLS for database connections
- [ ] Set network policies to restrict outbound traffic

### Sandboxed Workers

- [ ] NO credentials or secrets
- [ ] NO environment variables with sensitive data
- [ ] Verify `WORKER_TYPES=BLOCKING_SANDBOXED,NON_BLOCKING_SANDBOXED`
- [ ] Monitor resource usage (WASM overhead)
- [ ] Set memory limits (min 1GB per worker)

### General

- [ ] Enable Redis persistence (`--appendonly yes`)
- [ ] Use Redis authentication (`requirepass`)
- [ ] Enable TLS for Redis connections
- [ ] Set up network segmentation
- [ ] Configure firewall rules
- [ ] Enable audit logging
- [ ] Set up monitoring and alerts

## Troubleshooting

### Sandboxed Workers Not Processing Operations

**Check queue patterns:**
```bash
# Check if sandboxed queues have operations
redis-cli --scan --pattern "blazing:*:Station:*:AVAILABLESandboxed"
redis-cli LLEN "blazing:default:Station:{station_pk}:AVAILABLESandboxed"
```

**Check worker types:**
```bash
# Ensure sandboxed workers exist
redis-cli --scan --pattern "blazing:*:execution:WorkerThread:*" | \
  xargs -I {} redis-cli HGET {} "worker_type" | \
  sort | uniq -c
```

**Expected output:**
```
5 BLOCKING
10 NON_BLOCKING
2 BLOCKING_SANDBOXED
5 NON_BLOCKING_SANDBOXED
```

### High Memory Usage (Pyodide)

**Issue:** Each Pyodide worker uses ~50MB base + operation data

**Solutions:**
- Reduce number of sandboxed workers
- Increase memory limits: `memory: "2Gi"`
- Use pilot light to keep workers minimal when idle

### Service Calls from Sandbox Fail

**Check priorities:**
```bash
# Service calls should have priority = original + 100
redis-cli HGET "blazing:default:unit_definition:Operation:{op_pk}" "priority"
```

**Verify trusted workers exist:**
```bash
# Must have BLOCKING or NON_BLOCKING workers to execute services
redis-cli --scan --pattern "blazing:*:execution:WorkerThread:*" | \
  xargs -I {} redis-cli HGET {} "worker_type" | \
  grep -E "BLOCKING|NON_BLOCKING"
```

## Production Best Practices

### Resource Allocation

**Trusted Workers:**
- CPU: 500m - 1000m per worker
- Memory: 512Mi - 1Gi per worker
- Replicas: Scale based on queue depth

**Sandboxed Workers:**
- CPU: 500m - 1000m per worker (WASM overhead)
- Memory: 1Gi - 2Gi per worker (Pyodide + WASM heap)
- Replicas: Fewer needed (typically 30-50% of trusted)

### High Availability

```yaml
# Redis - use Redis Sentinel or Redis Cluster
services:
  redis-sentinel:
    image: redis:7-alpine
    command: redis-sentinel /etc/redis/sentinel.conf

# Foreman - single instance with restart policy
  foreman:
    restart: always
    deploy:
      replicas: 1

# Executors - multiple replicas
  executor:
    deploy:
      replicas: 5
  pyodide-executor:
    deploy:
      replicas: 3
```

### Logging

```yaml
# Centralized logging with Loki/Elasticsearch
services:
  executor:
    logging:
      driver: "json-file"
      options:
        max-size: "100m"
        max-file: "3"
        labels: "worker_type"
```
