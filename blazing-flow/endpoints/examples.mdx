Use these patterns to expose workflows as APIs without duplicating orchestration logic.

## Example 1: Basic HTTP Endpoint

```python
import asyncio
from blazing import Blazing
from blazing.web import create_asgi_app
import uvicorn

app = Blazing(
    api_url="https://api.blazingflow.com",
    api_token="<your-api-token>",
)

@app.endpoint(path="/calculate")
@app.workflow
async def calculate(x: int, y: int, services=None):
    return x + y

async def main():
    await app.publish()
    return await create_asgi_app(app)

fastapi_app = asyncio.run(main())
uvicorn.run(fastapi_app, host="0.0.0.0", port=8080)
```

## Example 2: Multi‑Step Pipeline

```python
from blazing import Blazing
from blazing.web import create_asgi_app
import asyncio

app = Blazing(api_url="https://api.blazingflow.com", api_token="<token>")

@app.step
async def validate(payload: dict, services=None):
    if "values" not in payload:
        raise ValueError("Missing values")
    return payload

@app.step
async def summarize(payload: dict, services=None):
    values = payload["values"]
    return {"count": len(values), "sum": sum(values)}

@app.endpoint(path="/analyze")
@app.workflow
async def analyze(payload: dict, services=None):
    payload = await validate(payload, services=services)
    return await summarize(payload, services=services)

async def main():
    await app.publish()
    return await create_asgi_app(app, title="Analysis API")

fastapi_app = asyncio.run(main())
```

## Example 3: Authenticated Endpoint

```python
import asyncio
from blazing import Blazing
from blazing.web import create_asgi_app
from fastapi.security import HTTPAuthorizationCredentials

app = Blazing(api_url="https://api.blazingflow.com", api_token="<token>")

async def verify_api_key(credentials: HTTPAuthorizationCredentials) -> bool:
    if not credentials:
        return False
    return credentials.credentials == "<expected-api-key>"

@app.endpoint(path="/secure/data", auth=verify_api_key)
@app.workflow
async def get_secure_data(user_id: int, services=None):
    return {"user_id": user_id, "data": "sensitive"}

async def main():
    await app.publish()
    return await create_asgi_app(app)

fastapi_app = asyncio.run(main())
```

## Example 4: Streaming Results

Use streaming when results are long‑running or you want incremental progress.

```python
import asyncio
from blazing import Blazing
from blazing.web import create_asgi_app

app = Blazing(api_url="https://api.blazingflow.com", api_token="<token>")

@app.endpoint(path="/stream", mode="stream")
@app.workflow
async def stream_numbers(n: int, services=None):
    for i in range(n):
        yield {"value": i}
        await asyncio.sleep(0.1)

async def main():
    await app.publish()
    return await create_asgi_app(app)

fastapi_app = asyncio.run(main())
```

## When to Use Endpoints vs Core Flow

Use Endpoints when you need HTTP/WebSocket interfaces. Use Core Flow when you only need direct Python calls or scheduled workflows.
