Configure Blazing Flow workflows, executors, retries, concurrency, and runtime behavior.

> **Note:** Most persistent infrastructure and scaling controls live in Blazing Core YAML.
> This page documents the optional `flow.yaml` configuration for workflow runtime behavior.

## Configuration File

Blazing Flow can use a `flow.yaml` file for workflow-specific runtime behavior.
Persistent infrastructure and scaling are configured in Blazing Core YAML.

```yaml
version: 1

flow:
  name: my-workflow
  description: "Production workflow for data processing"

  # Execution settings
  execution:
    timeout: 30m
    max_retries: 3
    retry_backoff: exponential

  # Concurrency settings
  concurrency:
    max_parallel_tasks: 10
    max_concurrent_executions: 100

  # Resource limits
  resources:
    cpu: 2
    memory: 4GB

  # State management
  state:
    storage: postgres
    ttl: 30d

  # Event configuration
  events:
    enabled: true
    webhook_url: "https://api.example.com/flow-events"
```

## Flow Settings

### Basic Configuration

```yaml
flow:
  # Workflow identifier (required)
  name: data-processing-workflow

  # Human-readable description
  description: "Process customer data and generate reports"

  # Version for tracking
  version: "1.2.0"

  # Tags for organization
  tags:
    - production
    - data-processing
    - daily

  # Environment
  environment: production  # development, staging, production
```

### Execution Settings

```yaml
flow:
  execution:
    # Global timeout for entire workflow
    timeout: 1h

    # Maximum retries for failed tasks
    max_retries: 3

    # Retry strategy
    retry_strategy: exponential  # exponential, linear, constant

    # Retry backoff
    retry_backoff:
      initial: 1s
      max: 5m
      multiplier: 2

    # Idempotency
    idempotent: true

    # Execution mode
    mode: async  # async, sync, deferred
```

### Concurrency Control

```yaml
flow:
  concurrency:
    # Maximum parallel tasks within single execution
    max_parallel_tasks: 10

    # Maximum concurrent workflow executions
    max_concurrent_executions: 100

    # Rate limiting
    rate_limit:
      requests_per_second: 50
      burst: 100

    # Task-specific concurrency
    task_limits:
      process_data:
        max_concurrent: 5
      send_email:
        max_concurrent: 20
```

### Resource Limits

```yaml
flow:
  resources:
    # CPU allocation
    cpu: 2  # cores

    # Memory allocation
    memory: 4GB

    # Disk space
    disk: 10GB

    # Network bandwidth
    network: 100Mbps

    # Per-task resource limits
    task_limits:
      heavy_computation:
        cpu: 4
        memory: 8GB
      light_task:
        cpu: 0.5
        memory: 512MB
```

## Task Configuration

### Task Defaults

```yaml
tasks:
  # Default settings for all tasks
  defaults:
    timeout: 5m
    retries: 3
    retry_backoff: exponential

    # Resource limits
    resources:
      cpu: 1
      memory: 2GB

    # Error handling
    on_failure: retry  # retry, skip, fail
```

### Task-Specific Configuration

```yaml
tasks:
  # Configure individual tasks
  process_data:
    timeout: 10m
    retries: 5
    priority: high  # high, normal, low
    resources:
      cpu: 2
      memory: 4GB

  send_notification:
    timeout: 30s
    retries: 2
    priority: low
    on_failure: skip  # Don't fail workflow if notification fails
```

## State Management

### State Storage

```yaml
state:
  # Storage backend
  storage: postgres  # memory, postgres, redis, dynamodb

  # Connection settings
  connection:
    host: db.example.com
    port: 5432
    database: flow_state
    user: flow_user
    password: "${DB_PASSWORD}"

  # State TTL
  ttl: 30d

  # Cleanup policy
  cleanup:
    enabled: true
    schedule: "0 2 * * *"  # Daily at 2 AM
    keep_completed: 7d
    keep_failed: 30d
```

### State Persistence

```yaml
state:
  # Persistence settings
  persistence:
    enabled: true

    # Checkpoint frequency
    # Note: Workflow checkpointing is not a Flow config option.

    # Snapshot configuration
    snapshots:
      enabled: true
      interval: 5m
      retention: 7d
```

## Event Configuration

### Event Publishing

```yaml
events:
  # Enable event publishing
  enabled: true

  # Event types to publish
  types:
    - workflow.started
    - workflow.completed
    - workflow.failed
    - task.started
    - task.completed
    - task.failed

  # Webhook delivery
  webhooks:
    - url: "https://api.example.com/events"
      headers:
        Authorization: "Bearer ${WEBHOOK_TOKEN}"
      retry:
        max_attempts: 3
        backoff: exponential

  # Event stream
  stream:
    enabled: true
    topic: "flow-events"
    partition_key: workflow_id
```

### Event Filtering

```yaml
events:
  # Filter events before publishing
  filters:
    # Only publish production events
    - field: environment
      operator: equals
      value: production

    # Only publish failed workflows
    - field: status
      operator: equals
      value: failed

    # Exclude certain task events
    - field: task_name
      operator: not_in
      values:
        - heartbeat
        - health_check
```

## Error Handling

### Retry Configuration

```yaml
error_handling:
  retries:
    # Retry strategy
    strategy: exponential

    # Retry configuration
    max_attempts: 5
    initial_delay: 1s
    max_delay: 5m
    multiplier: 2
    jitter: true

    # Retry conditions
    retry_on:
      - timeout
      - connection_error
      - rate_limit

    # Don't retry on these errors
    skip_on:
      - validation_error
      - authentication_error
```

### Fallback Handlers

```yaml
error_handling:
  fallbacks:
    # Task-level fallbacks
    - task: process_payment
      on_error: use_backup_processor
      max_fallback_attempts: 2

    # Workflow-level fallback
    - workflow: data_pipeline
      on_error: partial_completion
      notify: ops_team
```

### Dead Letter Queue

```yaml
error_handling:
  dead_letter:
    enabled: true

    # DLQ configuration
    queue: failed-workflows
    max_retry_attempts: 5

    # Routing
    routing:
      - condition: error_type == "validation"
        queue: validation-failures

      - condition: error_type == "timeout"
        queue: timeout-failures
```

## Monitoring Configuration

### Metrics

```yaml
monitoring:
  metrics:
    enabled: true

    # Export interval
    export_interval: 30s

    # Metric exporters
    exporters:
      - type: prometheus
        port: 9090
        path: /metrics

      - type: datadog
        api_key: "${DATADOG_API_KEY}"
        tags:
          - env:production
          - service:flow

      - type: cloudwatch
        namespace: Blazing/Flow
        region: us-east-1
```

### Logging

```yaml
monitoring:
  logging:
    # Log level
    level: info  # debug, info, warn, error

    # Log format
    format: json

    # Log outputs
    outputs:
      - type: stdout

      - type: file
        path: /var/log/flow.log
        rotation:
          max_size: 100MB
          max_age: 30d
          max_backups: 10

      - type: cloudwatch
        log_group: /blazing/flow
        log_stream: workflow-logs

    # Structured logging
    structured:
      enabled: true
      fields:
        - workflow_id
        - execution_id
        - task_name
        - environment
```

### Tracing

```yaml
monitoring:
  tracing:
    enabled: true

    # Trace sampler
    sampling:
      strategy: probabilistic
      rate: 0.1  # Sample 10% of traces

    # Trace exporter
    exporters:
      - type: jaeger
        endpoint: "http://jaeger:14268/api/traces"

      - type: zipkin
        endpoint: "http://zipkin:9411/api/v2/spans"

      - type: honeycomb
        api_key: "${HONEYCOMB_API_KEY}"
        dataset: blazing-flow
```

## Security Configuration

### Authentication

```yaml
security:
  authentication:
    # API authentication
    api:
      enabled: true
      type: bearer  # bearer, basic, oauth2
      token_validation:
        endpoint: "https://auth.example.com/validate"

    # Service-to-service auth
    service:
      enabled: true
      type: mtls
      certificates:
        ca: /path/to/ca.pem
        cert: /path/to/cert.pem
        key: /path/to/key.pem
```

### Authorization

```yaml
security:
  authorization:
    # Role-based access control
    rbac:
      enabled: true
      roles:
        admin:
          - workflow.create
          - workflow.delete
          - workflow.execute

        developer:
          - workflow.create
          - workflow.execute

        operator:
          - workflow.execute
          - workflow.view
```

### Secrets Management

```yaml
security:
  secrets:
    # Secrets backend
    backend: vault  # vault, aws_secrets, azure_keyvault

    # Vault configuration
    vault:
      address: "https://vault.example.com"
      token: "${VAULT_TOKEN}"
      path: secret/blazing/flow

    # Secret rotation
    rotation:
      enabled: true
      interval: 30d
```

## Network Configuration

### HTTP Client

```yaml
network:
  http:
    # Timeout settings
    timeout:
      connect: 10s
      read: 30s
      write: 30s

    # Retry settings
    retries:
      max_attempts: 3
      backoff: exponential

    # Connection pooling
    pool:
      max_idle_connections: 100
      max_connections_per_host: 10
      idle_timeout: 90s

    # TLS configuration
    tls:
      enabled: true
      verify: true
      min_version: "1.2"
```

### Rate Limiting

```yaml
network:
  rate_limiting:
    # Global rate limits
    global:
      requests_per_second: 100
      burst: 200

    # Per-endpoint limits
    endpoints:
      "api.stripe.com":
        requests_per_second: 50
        burst: 100

      "api.external.com":
        requests_per_second: 10
        burst: 20
```

## Environment Variables

Override configuration with environment variables:

```bash
# Flow settings
FLOW_NAME=my-workflow
FLOW_ENVIRONMENT=production

# Execution settings
FLOW_TIMEOUT=30m
FLOW_MAX_RETRIES=5

# State storage
FLOW_STATE_STORAGE=postgres
FLOW_STATE_CONNECTION=postgresql://localhost/flow

# Monitoring
FLOW_METRICS_ENABLED=true
FLOW_METRICS_PORT=9090

# Security
FLOW_AUTH_ENABLED=true
FLOW_AUTH_TOKEN=secret-token
```

## Complete Example

Here's a complete production configuration:

```yaml
version: 1

flow:
  name: production-workflow
  version: "2.1.0"
  environment: production

  execution:
    timeout: 1h
    max_retries: 5
    retry_strategy: exponential

  concurrency:
    max_parallel_tasks: 20
    max_concurrent_executions: 100
    rate_limit:
      requests_per_second: 50

  resources:
    cpu: 4
    memory: 8GB

state:
  storage: postgres
  connection:
    host: db.example.com
    database: flow_production
  ttl: 30d

events:
  enabled: true
  webhooks:
    - url: "https://api.example.com/events"

error_handling:
  retries:
    max_attempts: 5
    initial_delay: 1s
    max_delay: 5m
  dead_letter:
    enabled: true
    queue: failed-workflows

monitoring:
  metrics:
    enabled: true
    exporters:
      - type: prometheus
        port: 9090
  logging:
    level: info
    format: json
  tracing:
    enabled: true
    sampling:
      rate: 0.1

security:
  authentication:
    api:
      enabled: true
      type: bearer
  secrets:
    backend: vault

network:
  http:
    timeout:
      connect: 10s
      read: 30s
    pool:
      max_connections: 100
  rate_limiting:
    global:
      requests_per_second: 100
```
