These examples show core Blazing Flow usage without the HTTP layer.

## Example 1: Simple Step + Workflow

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def hello(name: str, services=None):
        return f"Hello, {name}!"

    @app.workflow
    async def greet(name: str, services=None):
        return await hello(name, services=services)

    await app.publish()
    result = await app.greet("World").wait_result()
    print(result)

if __name__ == "__main__":
    asyncio.run(main())
```

## Example 2: Fan‑Out / Fan‑In

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def square(x: int, services=None):
        return x * x

    @app.workflow
    async def squares(nums: list[int], services=None):
        tasks = [square(n, services=services) for n in nums]
        results = await asyncio.gather(*[t.result() for t in tasks])
        return results

    await app.publish()
    print(await app.squares([1, 2, 3, 4]).wait_result())

if __name__ == "__main__":
    asyncio.run(main())
```

## Example 3: Service + Connector

```python
import asyncio
from blazing import Blazing
from blazing.base import BaseService

async def main():
    app = Blazing()

    @app.service
    class UserStore(BaseService):
        def __init__(self, connectors):
            self._db = connectors.get("postgres")

        async def get_user(self, user_id: int) -> dict:
            row = await self._db.fetch_one(
                "SELECT * FROM users WHERE id = :id", {"id": user_id}
            )
            return dict(row)

    @app.workflow
    async def fetch_user(user_id: int, services=None):
        return await services["UserStore"].get_user(user_id)

    await app.publish()

if __name__ == "__main__":
    asyncio.run(main())
```

## Example 4: Dynamic Batching

```python
import asyncio
from blazing import Blazing, batched, BatchConfig

async def main():
    app = Blazing()

    @batched(BatchConfig(max_batch_size=16, wait_ms=50))
    @app.step
    async def embed(texts: list[str], services=None):
        return [t.upper() for t in texts]

    @app.workflow
    async def run(text: str, services=None):
        return await embed(text, services=services)

    await app.publish()

if __name__ == "__main__":
    asyncio.run(main())
```

## Example 5: Scheduled Workflow

```python
import asyncio
from blazing import Blazing, Cron

async def main():
    app = Blazing()

    @app.workflow(schedule=Cron("0 * * * *"))
    async def hourly_job(services=None):
        return {"status": "ok"}

    await app.publish()

if __name__ == "__main__":
    asyncio.run(main())
```
