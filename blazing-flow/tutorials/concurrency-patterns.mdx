## Concurrency Control with Semaphore

### Basic Semaphore Usage

Use `Semaphore` from the Blazing SDK to limit concurrent operations:

```python
import asyncio
from blazing import Blazing, Semaphore

async def main():
    app = Blazing()

    # Create semaphore with limit of 3 concurrent operations
    sem = Semaphore(3)

    @app.step
    async def rate_limited_operation(item_id: int, services=None):
        """Operation with concurrency limit."""
        async with sem:
            # Only 3 of these can run at once
            await asyncio.sleep(1)  # Simulate work
            return f"Processed item {item_id}"

    @app.workflow
    async def batch_process(items: list, services=None):
        """Process items with rate limiting."""
        tasks = [
            rate_limited_operation(item, services=services)
            for item in items
        ]
        results = await asyncio.gather(*tasks)
        return results

    await app.publish()

    # Process 10 items with max 3 concurrent
    result = await app.batch_process(list(range(10))).wait_result()
    print(f"Processed {len(result)} items")

if __name__ == "__main__":
    asyncio.run(main())
```

## Dynamic Batching

Batch individual calls into a single execution with `@batched`:

```python
import asyncio
from blazing import Blazing, batched, BatchConfig

async def main():
    app = Blazing()

    @batched(BatchConfig(max_batch_size=16, wait_ms=25))
    @app.step
    async def classify(texts: list[str], services=None) -> list[str]:
        return await services["Classifier"].predict(texts)

    @app.workflow
    async def run_batch(inputs: list[str], services=None):
        tasks = [classify(t, services=services) for t in inputs]
        return await asyncio.gather(*tasks)

    await app.publish()

    inputs = [f"item-{i}" for i in range(100)]
    result = await app.run_batch(inputs).wait_result()
    print(f"Processed {len(result)} items")

if __name__ == "__main__":
    asyncio.run(main())
```

**Key points:**
- Import `Semaphore` from `blazing` (NOT `asyncio.Semaphore`)
- Use `async with sem:` to acquire/release automatically
- Limits concurrent executions to specified count

### Rate Limiting API Calls

Prevent overwhelming external APIs:

```python
import asyncio
from blazing import Blazing, Semaphore

async def main():
    app = Blazing()

    # Allow only 5 concurrent API calls
    api_semaphore = Semaphore(5)

    @app.step
    async def fetch_from_api(url: str, services=None):
        """Fetch data with rate limiting."""
        async with api_semaphore:
            # Simulate API call
            await asyncio.sleep(0.5)
            return {"url": url, "data": f"Data from {url}"}

    @app.workflow
    async def fetch_multiple_urls(urls: list, services=None):
        """Fetch from multiple URLs with rate limiting."""
        tasks = [
            fetch_from_api(url, services=services)
            for url in urls
        ]
        results = await asyncio.gather(*tasks)
        return {
            "total": len(results),
            "results": results
        }

    await app.publish()

    # Fetch 20 URLs with max 5 concurrent
    urls = [f"https://api.example.com/data/{i}" for i in range(20)]
    result = await app.fetch_multiple_urls(urls).wait_result()
    print(f"Fetched {result['total']} URLs")

if __name__ == "__main__":
    asyncio.run(main())
```

### Resource Pool Management

Manage limited resources like database connections:

```python
import asyncio
from blazing import Blazing, Semaphore

async def main():
    app = Blazing()

    # Pool of 10 database connections
    db_pool = Semaphore(10)

    @app.step
    async def query_database(query: str, services=None):
        """Execute query with connection pooling."""
        async with db_pool:
            # Acquire connection from pool
            await asyncio.sleep(0.1)  # Simulate query
            return {"query": query, "rows": 42}

    @app.workflow
    async def batch_queries(queries: list, services=None):
        """Execute multiple queries with pooling."""
        tasks = [
            query_database(query, services=services)
            for query in queries
        ]
        results = await asyncio.gather(*tasks)
        return {"executed": len(results), "results": results}

    await app.publish()

    queries = [f"SELECT * FROM table_{i}" for i in range(50)]
    result = await app.batch_queries(queries).wait_result()
    print(f"Executed {result['executed']} queries")

if __name__ == "__main__":
    asyncio.run(main())
```

## Parallel Workflow Execution

### Running Multiple Workflow Instances

Execute multiple workflow instances concurrently:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process_item(item: dict, services=None):
        """Process a single item."""
        await asyncio.sleep(0.5)
        return {
            "id": item["id"],
            "processed": True,
            "value": item["value"] * 2
        }

    @app.workflow
    async def item_workflow(item: dict, services=None):
        """Workflow for processing one item."""
        result = await process_item(item, services=services)
        return result

    await app.publish()

    # Create multiple workflow instances
    items = [{"id": i, "value": i * 10} for i in range(20)]

    # Execute all workflows in parallel
    tasks = [
        app.item_workflow(item).wait_result()
        for item in items
    ]
    results = await asyncio.gather(*tasks)

    print(f"Processed {len(results)} items in parallel")

if __name__ == "__main__":
    asyncio.run(main())
```

### Controlled Parallel Execution

Combine Semaphore with parallel workflows for controlled concurrency:

```python
import asyncio
from blazing import Blazing, Semaphore

async def main():
    app = Blazing()

    # Limit to 5 concurrent workflows
    workflow_sem = Semaphore(5)

    @app.step
    async def expensive_operation(data: dict, services=None):
        """Resource-intensive operation."""
        await asyncio.sleep(2)
        return {"processed": data, "result": "success"}

    @app.workflow
    async def controlled_workflow(data: dict, services=None):
        """Workflow with concurrency control."""
        async with workflow_sem:
            result = await expensive_operation(data, services=services)
            return result

    await app.publish()

    # Process 100 items with max 5 concurrent workflows
    items = [{"id": i, "value": i} for i in range(100)]

    tasks = [
        app.controlled_workflow(item).wait_result()
        for item in items
    ]
    results = await asyncio.gather(*tasks)

    print(f"Completed {len(results)} workflows")

if __name__ == "__main__":
    asyncio.run(main())
```

## Advanced Patterns

### Fan-Out / Fan-In

Distribute work across multiple steps, then aggregate:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process_chunk(chunk: list, chunk_id: int, services=None):
        """Process a data chunk."""
        await asyncio.sleep(0.5)
        return {"chunk_id": chunk_id, "count": len(chunk), "sum": sum(chunk)}

    @app.step
    async def aggregate_results(results: list, services=None):
        """Aggregate all chunk results."""
        total_count = sum(r["count"] for r in results)
        total_sum = sum(r["sum"] for r in results)
        return {"total_count": total_count, "total_sum": total_sum}

    @app.workflow
    async def fan_out_fan_in(data: list, chunk_size: int, services=None):
        """Fan-out: process chunks, fan-in: aggregate."""
        # Fan-out: split data into chunks
        chunks = [
            data[i:i + chunk_size]
            for i in range(0, len(data), chunk_size)
        ]

        # Process all chunks in parallel
        chunk_tasks = [
            process_chunk(chunk, i, services=services)
            for i, chunk in enumerate(chunks)
        ]
        chunk_results = await asyncio.gather(*chunk_tasks)

        # Fan-in: aggregate results
        final_result = await aggregate_results(chunk_results, services=services)
        return final_result

    await app.publish()

    # Process 1000 items in chunks of 100
    data = list(range(1000))
    result = await app.fan_out_fan_in(data, chunk_size=100).wait_result()
    print(f"Total count: {result['total_count']}")
    print(f"Total sum: {result['total_sum']}")

if __name__ == "__main__":
    asyncio.run(main())
```

### Pipeline Processing

Chain multiple processing stages:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def extract(source: str, services=None):
        """Extract data from source."""
        await asyncio.sleep(0.2)
        return {"source": source, "raw_data": [1, 2, 3, 4, 5]}

    @app.step
    async def transform(data: dict, services=None):
        """Transform extracted data."""
        await asyncio.sleep(0.2)
        transformed = [x * 2 for x in data["raw_data"]]
        return {"source": data["source"], "transformed": transformed}

    @app.step
    async def load(data: dict, services=None):
        """Load transformed data."""
        await asyncio.sleep(0.2)
        return {"source": data["source"], "loaded": True, "count": len(data["transformed"])}

    @app.workflow
    async def etl_pipeline(sources: list, services=None):
        """Execute ETL pipeline for multiple sources."""
        # Stage 1: Extract
        extract_tasks = [extract(source, services=services) for source in sources]
        extracted = await asyncio.gather(*extract_tasks)

        # Stage 2: Transform
        transform_tasks = [transform(data, services=services) for data in extracted]
        transformed = await asyncio.gather(*transform_tasks)

        # Stage 3: Load
        load_tasks = [load(data, services=services) for data in transformed]
        loaded = await asyncio.gather(*load_tasks)

        return {"pipeline_complete": True, "sources_processed": len(loaded)}

    await app.publish()

    sources = [f"source_{i}" for i in range(10)]
    result = await app.etl_pipeline(sources).wait_result()
    print(f"Processed {result['sources_processed']} sources")

if __name__ == "__main__":
    asyncio.run(main())
```

### Batched Processing with Windows

Process data in sliding windows:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process_window(window: list, window_id: int, services=None):
        """Process a time window of data."""
        await asyncio.sleep(0.3)
        return {
            "window_id": window_id,
            "count": len(window),
            "avg": sum(window) / len(window) if window else 0
        }

    @app.workflow
    async def windowed_processing(data: list, window_size: int, services=None):
        """Process data in sliding windows."""
        windows = []
        for i in range(0, len(data), window_size):
            window = data[i:i + window_size]
            windows.append(window)

        # Process all windows in parallel
        tasks = [
            process_window(window, i, services=services)
            for i, window in enumerate(windows)
        ]
        results = await asyncio.gather(*tasks)

        return {
            "windows_processed": len(results),
            "results": results
        }

    await app.publish()

    # Process 100 data points in windows of 10
    data = list(range(100))
    result = await app.windowed_processing(data, window_size=10).wait_result()
    print(f"Processed {result['windows_processed']} windows")

if __name__ == "__main__":
    asyncio.run(main())
```

## Performance Optimization

### Optimal Concurrency Levels

Finding the right concurrency limit:

```python
import asyncio
import time
from blazing import Blazing, Semaphore

async def main():
    app = Blazing()

    async def benchmark_concurrency(limit: int):
        """Benchmark a specific concurrency limit."""
        sem = Semaphore(limit)

        @app.step
        async def work(item: int, services=None):
            async with sem:
                await asyncio.sleep(0.1)
                return item * 2

        @app.workflow
        async def benchmark_workflow(items: list, services=None):
            tasks = [work(item, services=services) for item in items]
            results = await asyncio.gather(*tasks)
            return len(results)

        await app.publish()

        start = time.time()
        result = await app.benchmark_workflow(list(range(100))).wait_result()
        elapsed = time.time() - start

        return {"limit": limit, "time": elapsed, "throughput": result / elapsed}

    # Test different concurrency limits
    for limit in [1, 5, 10, 20, 50]:
        result = await benchmark_concurrency(limit)
        print(f"Limit {result['limit']}: {result['time']:.2f}s, {result['throughput']:.1f} ops/s")

if __name__ == "__main__":
    asyncio.run(main())
```

## Best Practices

1. **Use Blazing Semaphore** - Import from `blazing`, not `asyncio`
2. **Set appropriate limits** - Match resource capacity (DB connections, API rate limits)
3. **Always use `async with`** - Ensures semaphore is released even on exceptions
4. **Monitor queue depth** - Track how many tasks are waiting
5. **Tune concurrency** - Benchmark to find optimal settings
6. **Avoid deadlocks** - Don't nest semaphore acquisitions
7. **Use timeouts** - Prevent tasks from holding semaphore indefinitely

## Common Patterns

### Rate Limiting
```python
# 10 requests per second
sem = Semaphore(10)
async with sem:
    await api_call()
```

### Resource Pooling
```python
# Pool of 20 database connections
db_pool = Semaphore(20)
async with db_pool:
    await db.query()
```

### Workflow Throttling
```python
# Max 5 concurrent workflows
workflow_sem = Semaphore(5)
async with workflow_sem:
    result = await expensive_workflow()
```
