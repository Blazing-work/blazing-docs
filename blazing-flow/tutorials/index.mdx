## Quick Reference

### Basic Workflow Pattern

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process(x: int, services=None):
        """Process a single value."""
        return x * 2

    @app.workflow
    async def my_workflow(value: int, services=None):
        """Complete workflow orchestration."""
        result = await process(value, services=services)
        return result

    await app.publish()

    # Execute workflow
    result = await app.my_workflow(5).wait_result()
    print(f"Result: {result}")  # Result: 10

if __name__ == "__main__":
    asyncio.run(main())
```

### Parallel Execution Pattern

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process(x: int, services=None):
        return x * 2

    @app.workflow
    async def my_workflow(value: int, services=None):
        result = await process(value, services=services)
        return result

    await app.publish()

    # Execute multiple workflows in parallel
    tasks = [
        app.my_workflow(i).wait_result()
        for i in range(10)
    ]
    results = await asyncio.gather(*tasks)
    print(f"Processed {len(results)} workflows")

if __name__ == "__main__":
    asyncio.run(main())
```

### Error Handling Pattern

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def risky_operation(x: int, services=None):
        if x < 0:
            raise ValueError("Value must be non-negative")
        return x * 2

    @app.workflow
    async def safe_workflow(value: int, services=None):
        try:
            result = await risky_operation(value, services=services)
            return {"success": True, "result": result}
        except ValueError as e:
            return {"success": False, "error": str(e)}

    await app.publish()

    # Handle both success and failure
    result = await app.safe_workflow(-5).wait_result()
    print(result)  # {'success': False, 'error': 'Value must be non-negative'}

if __name__ == "__main__":
    asyncio.run(main())
```
