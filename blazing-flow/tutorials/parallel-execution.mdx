Learn how to execute multiple workflow instances in parallel to maximize performance and throughput with Blazing Flow.

## Basic Parallel Execution

### Running Multiple Workflows

Execute multiple workflow instances concurrently using `asyncio.gather`:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process_data(value: int, services=None):
        """Process a single value."""
        await asyncio.sleep(0.5)  # Simulate work
        return value * 2

    @app.workflow
    async def data_workflow(value: int, services=None):
        """Workflow for processing one value."""
        result = await process_data(value, services=services)
        return {"input": value, "output": result}

    await app.publish()

    # Execute multiple workflows in parallel
    tasks = [
        app.data_workflow(i).wait_result()
        for i in range(10)
    ]
    results = await asyncio.gather(*tasks)

    print(f"Processed {len(results)} workflows in parallel")
    for result in results[:3]:
        print(f"  {result['input']} -> {result['output']}")

if __name__ == "__main__":
    asyncio.run(main())
```

**Output:**
```
Processed 10 workflows in parallel
  0 -> 0
  1 -> 2
  2 -> 4
```

## Performance Benefits

### Sequential vs Parallel Comparison

**Sequential execution** (10 workflows × 0.5s each = ~5 seconds):

```python
import asyncio
import time
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def slow_step(value: int, services=None):
        await asyncio.sleep(0.5)
        return value * 2

    @app.workflow
    async def slow_workflow(value: int, services=None):
        result = await slow_step(value, services=services)
        return result

    await app.publish()

    # SEQUENTIAL - slow!
    start = time.time()
    results = []
    for i in range(10):
        result = await app.slow_workflow(i).wait_result()
        results.append(result)
    elapsed = time.time() - start

    print(f"Sequential: {len(results)} workflows in {elapsed:.2f}s")
    # Output: Sequential: 10 workflows in ~5.0s

if __name__ == "__main__":
    asyncio.run(main())
```

**Parallel execution** (10 workflows × 0.5s each = ~0.5 seconds):

```python
import asyncio
import time
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def fast_step(value: int, services=None):
        await asyncio.sleep(0.5)
        return value * 2

    @app.workflow
    async def fast_workflow(value: int, services=None):
        result = await fast_step(value, services=services)
        return result

    await app.publish()

    # PARALLEL - fast!
    start = time.time()
    tasks = [app.fast_workflow(i).wait_result() for i in range(10)]
    results = await asyncio.gather(*tasks)
    elapsed = time.time() - start

    print(f"Parallel: {len(results)} workflows in {elapsed:.2f}s")
    print(f"Speedup: {(10 * 0.5) / elapsed:.1f}x")
    # Output: Parallel: 10 workflows in ~0.5s
    # Speedup: 10.0x

if __name__ == "__main__":
    asyncio.run(main())
```

## Real-World Examples

### Batch Data Processing

Process multiple data items in parallel:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def fetch_data(item_id: str, services=None):
        """Fetch data for one item."""
        await asyncio.sleep(0.3)
        return {"id": item_id, "data": f"Data for {item_id}"}

    @app.step
    async def transform_data(data: dict, services=None):
        """Transform fetched data."""
        await asyncio.sleep(0.2)
        return {
            "id": data["id"],
            "transformed": data["data"].upper()
        }

    @app.workflow
    async def process_item(item_id: str, services=None):
        """Complete processing pipeline for one item."""
        data = await fetch_data(item_id, services=services)
        result = await transform_data(data, services=services)
        return result

    await app.publish()

    # Process 50 items in parallel
    import time
    start = time.time()

    items = [f"item_{i}" for i in range(50)]
    tasks = [app.process_item(item).wait_result() for item in items]
    results = await asyncio.gather(*tasks)

    elapsed = time.time() - start
    print(f"Processed {len(results)} items in {elapsed:.2f}s")
    print(f"Throughput: {len(results) / elapsed:.1f} items/s")

if __name__ == "__main__":
    asyncio.run(main())
```

### API Data Aggregation

Fetch data from multiple sources concurrently:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def fetch_api_data(source: str, services=None):
        """Fetch from one API source."""
        await asyncio.sleep(0.5)  # Simulate API call
        return {
            "source": source,
            "data": f"Response from {source}",
            "timestamp": "2024-01-01T12:00:00Z"
        }

    @app.workflow
    async def aggregate_sources(sources: list, services=None):
        """Fetch from all sources in parallel."""
        # Create parallel fetch tasks
        tasks = [
            fetch_api_data(source, services=services)
            for source in sources
        ]

        # Fetch all concurrently
        results = await asyncio.gather(*tasks)

        return {
            "sources_fetched": len(results),
            "data": results
        }

    await app.publish()

    sources = [
        "api.weather.com",
        "api.stocks.com",
        "api.news.com",
        "api.crypto.com"
    ]

    result = await app.aggregate_sources(sources).wait_result()
    print(f"Fetched from {result['sources_fetched']} sources")

if __name__ == "__main__":
    asyncio.run(main())
```

## Scaling Patterns

### Batch Processing

Process data in manageable batches:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process_item(item: dict, services=None):
        """Process a single item."""
        await asyncio.sleep(0.1)
        return {"id": item["id"], "processed": True}

    @app.workflow
    async def process_batch(batch: list, services=None):
        """Process one batch of items."""
        tasks = [
            process_item(item, services=services)
            for item in batch
        ]
        results = await asyncio.gather(*tasks)
        return {"batch_size": len(results), "results": results}

    await app.publish()

    # Process 1000 items in batches of 100
    all_items = [{"id": i, "value": i} for i in range(1000)]
    batch_size = 100

    batches = [
        all_items[i:i + batch_size]
        for i in range(0, len(all_items), batch_size)
    ]

    # Process batches in parallel
    tasks = [app.process_batch(batch).wait_result() for batch in batches]
    results = await asyncio.gather(*tasks)

    total_processed = sum(r["batch_size"] for r in results)
    print(f"Processed {total_processed} items in {len(results)} batches")

if __name__ == "__main__":
    asyncio.run(main())
```

### Dynamic Workload Distribution

Distribute work dynamically based on workload:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def light_task(task_id: int, services=None):
        """Quick task."""
        await asyncio.sleep(0.1)
        return {"id": task_id, "type": "light", "result": task_id * 2}

    @app.step
    async def heavy_task(task_id: int, services=None):
        """Slow task."""
        await asyncio.sleep(1.0)
        return {"id": task_id, "type": "heavy", "result": task_id ** 2}

    @app.workflow
    async def mixed_workload(light_count: int, heavy_count: int, services=None):
        """Process mix of light and heavy tasks."""
        # Create all tasks
        light_tasks = [
            light_task(i, services=services)
            for i in range(light_count)
        ]
        heavy_tasks = [
            heavy_task(i, services=services)
            for i in range(heavy_count)
        ]

        # Execute all in parallel
        all_tasks = light_tasks + heavy_tasks
        results = await asyncio.gather(*all_tasks)

        return {
            "light_count": light_count,
            "heavy_count": heavy_count,
            "total_processed": len(results)
        }

    await app.publish()

    import time
    start = time.time()

    result = await app.mixed_workload(
        light_count=100,
        heavy_count=5
    ).wait_result()

    elapsed = time.time() - start
    print(f"Processed {result['total_processed']} tasks in {elapsed:.2f}s")

if __name__ == "__main__":
    asyncio.run(main())
```

## Advanced Techniques

### Map-Reduce Pattern

Parallel processing with aggregation:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def map_step(value: int, services=None):
        """Map: transform a value."""
        await asyncio.sleep(0.1)
        return value ** 2

    @app.step
    async def reduce_step(values: list, services=None):
        """Reduce: aggregate all values."""
        return {
            "count": len(values),
            "sum": sum(values),
            "avg": sum(values) / len(values) if values else 0
        }

    @app.workflow
    async def map_reduce(data: list, services=None):
        """Map-reduce pattern."""
        # Map phase: process all items in parallel
        map_tasks = [map_step(value, services=services) for value in data]
        mapped = await asyncio.gather(*map_tasks)

        # Reduce phase: aggregate results
        result = await reduce_step(mapped, services=services)
        return result

    await app.publish()

    data = list(range(100))
    result = await app.map_reduce(data).wait_result()

    print(f"Processed {result['count']} values")
    print(f"Sum: {result['sum']}")
    print(f"Average: {result['avg']:.2f}")

if __name__ == "__main__":
    asyncio.run(main())
```

### Hierarchical Parallelism

Nested parallel execution:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process_item(item: int, services=None):
        """Process one item."""
        await asyncio.sleep(0.1)
        return item * 2

    @app.workflow
    async def process_group(group: list, services=None):
        """Process a group of items in parallel."""
        tasks = [process_item(item, services=services) for item in group]
        results = await asyncio.gather(*tasks)
        return {"group_size": len(results), "sum": sum(results)}

    @app.workflow
    async def process_all_groups(groups: list, services=None):
        """Process multiple groups in parallel."""
        # Parallel across groups, each group processes items in parallel
        tasks = [process_group(group, services=services) for group in groups]
        results = await asyncio.gather(*tasks)

        total_sum = sum(r["sum"] for r in results)
        return {
            "groups_processed": len(results),
            "total_sum": total_sum
        }

    await app.publish()

    # Create groups of items
    groups = [
        list(range(i * 10, (i + 1) * 10))
        for i in range(10)
    ]

    result = await app.process_all_groups(groups).wait_result()
    print(f"Processed {result['groups_processed']} groups")
    print(f"Total sum: {result['total_sum']}")

if __name__ == "__main__":
    asyncio.run(main())
```

## Best Practices

1. **Create all tasks first** - Build the task list before awaiting
2. **Use asyncio.gather** - Efficient parallel execution primitive
3. **Batch large workloads** - Process in chunks for better control
4. **Monitor resource usage** - Watch memory and network bandwidth
5. **Set concurrency limits** - Use Semaphore when needed (see [Concurrency Patterns](/docs/tutorials/concurrency-patterns))
6. **Handle failures gracefully** - Some workflows may fail (see [Error Handling](/docs/tutorials/error-handling))

## Common Patterns

### All workflows in parallel
```python
tasks = [app.workflow(i).wait_result() for i in range(100)]
results = await asyncio.gather(*tasks)
```

### Batched processing
```python
for batch in batches:
    tasks = [app.workflow(item).wait_result() for item in batch]
    results = await asyncio.gather(*tasks)
```

### Map-reduce
```python
# Map
mapped = await asyncio.gather(*[map_fn(x) for x in data])
# Reduce
result = await reduce_fn(mapped)
```
