## Common Error Scenarios

### 1. Parameter Validation

Validate parameters before processing to catch errors early:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def process_value(x: int, services=None):
        """Process a value with validation."""
        if x < 0:
            raise ValueError("Value must be non-negative")
        if x > 1000:
            raise ValueError("Value too large (max: 1000)")
        return x * 2

    @app.workflow
    async def safe_workflow(value: int, services=None):
        """Workflow with parameter validation."""
        try:
            result = await process_value(value, services=services)
            return {"success": True, "result": result}
        except ValueError as e:
            return {"success": False, "error": str(e)}

    await app.publish()

    # Test with invalid input
    result = await app.safe_workflow(-5).wait_result()
    print(result)
    # Output: {'success': False, 'error': 'Value must be non-negative'}

if __name__ == "__main__":
    asyncio.run(main())
```

### 2. Missing Required Data

Handle cases where required data is unavailable:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def fetch_user_data(user_id: str, services=None):
        """Fetch user data from database."""
        # Simulate database lookup
        if user_id not in ["user1", "user2"]:
            raise KeyError(f"User {user_id} not found")
        return {"id": user_id, "name": f"User {user_id}"}

    @app.workflow
    async def get_user_profile(user_id: str, services=None):
        """Get user profile with error handling."""
        try:
            data = await fetch_user_data(user_id, services=services)
            return {"success": True, "data": data}
        except KeyError as e:
            return {"success": False, "error": str(e)}

    await app.publish()

    # Test with missing user
    result = await app.get_user_profile("user999").wait_result()
    print(result)
    # Output: {'success': False, 'error': 'User user999 not found'}

if __name__ == "__main__":
    asyncio.run(main())
```

### 3. Type Validation

Ensure correct types are provided:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def calculate_total(items: list, services=None):
        """Calculate total with type checking."""
        if not isinstance(items, list):
            raise TypeError(f"Expected list, got {type(items).__name__}")
        if not all(isinstance(x, (int, float)) for x in items):
            raise TypeError("All items must be numbers")
        return sum(items)

    @app.workflow
    async def safe_calculation(items, services=None):
        """Calculate with type validation."""
        try:
            total = await calculate_total(items, services=services)
            return {"success": True, "total": total}
        except TypeError as e:
            return {"success": False, "error": str(e)}

    await app.publish()

    # Test with invalid type
    result = await app.safe_calculation("not a list").wait_result()
    print(result)
    # Output: {'success': False, 'error': 'Expected list, got str'}

if __name__ == "__main__":
    asyncio.run(main())
```

## Handling Step Failures

### Try-Except in Workflows

Workflows can catch and handle step failures:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def risky_operation(x: int, services=None):
        """Operation that might fail."""
        if x == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        return 100 / x

    @app.step
    async def safe_fallback(services=None):
        """Fallback operation."""
        return 0  # Safe default value

    @app.workflow
    async def resilient_workflow(value: int, services=None):
        """Workflow with error recovery."""
        try:
            result = await risky_operation(value, services=services)
            return {"status": "success", "result": result}
        except ZeroDivisionError:
            # Use fallback on error
            fallback = await safe_fallback(services=services)
            return {"status": "fallback", "result": fallback}

    await app.publish()

    # Test with failing input
    result = await app.resilient_workflow(0).wait_result()
    print(result)
    # Output: {'status': 'fallback', 'result': 0}

if __name__ == "__main__":
    asyncio.run(main())
```

### Graceful Degradation

Return partial results when some operations fail:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def fetch_data(source: str, services=None):
        """Fetch data from a source."""
        if source == "bad_source":
            raise ConnectionError(f"Cannot connect to {source}")
        return {"source": source, "data": f"Data from {source}"}

    @app.workflow
    async def multi_source_fetch(sources: list, services=None):
        """Fetch from multiple sources with graceful degradation."""
        results = []
        errors = []

        for source in sources:
            try:
                data = await fetch_data(source, services=services)
                results.append(data)
            except ConnectionError as e:
                errors.append({"source": source, "error": str(e)})

        return {
            "successes": results,
            "failures": errors,
            "success_rate": len(results) / len(sources)
        }

    await app.publish()

    # Test with mixed sources
    result = await app.multi_source_fetch(
        ["source1", "bad_source", "source3"]
    ).wait_result()
    print(f"Successes: {len(result['successes'])}")
    print(f"Failures: {len(result['failures'])}")
    print(f"Success rate: {result['success_rate']:.1%}")

if __name__ == "__main__":
    asyncio.run(main())
```

## Timeout Handling

### Step-Level Timeouts

Handle operations that might take too long:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def slow_operation(delay: int, services=None):
        """Operation that takes variable time."""
        await asyncio.sleep(delay)
        return f"Completed after {delay}s"

    @app.workflow
    async def workflow_with_timeout(delay: int, services=None):
        """Workflow with timeout handling."""
        try:
            # Set timeout for the operation
            result = await asyncio.wait_for(
                slow_operation(delay, services=services),
                timeout=5.0
            )
            return {"success": True, "result": result}
        except asyncio.TimeoutError:
            return {"success": False, "error": "Operation timed out after 5s"}

    await app.publish()

    # Test with slow operation
    result = await app.workflow_with_timeout(10).wait_result()
    print(result)
    # Output: {'success': False, 'error': 'Operation timed out after 5s'}

if __name__ == "__main__":
    asyncio.run(main())
```

## Retry Patterns

### Automatic Retry with Exponential Backoff

Implement retry logic for transient failures:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def unreliable_operation(attempt_count: int = 0, services=None):
        """Operation that fails sometimes."""
        # Simulate 70% failure rate
        import random
        if random.random() < 0.7:
            raise ConnectionError("Network error")
        return "Success"

    @app.workflow
    async def workflow_with_retry(services=None):
        """Workflow with automatic retry."""
        max_retries = 3

        for attempt in range(max_retries):
            try:
                result = await unreliable_operation(
                    attempt_count=attempt,
                    services=services
                )
                return {
                    "success": True,
                    "result": result,
                    "attempts": attempt + 1
                }
            except ConnectionError as e:
                if attempt == max_retries - 1:
                    # Final attempt failed
                    return {
                        "success": False,
                        "error": str(e),
                        "attempts": max_retries
                    }

                # Exponential backoff
                wait_time = 2 ** attempt
                await asyncio.sleep(wait_time)

    await app.publish()

    result = await app.workflow_with_retry().wait_result()
    print(f"Success: {result['success']}")
    print(f"Attempts: {result['attempts']}")

if __name__ == "__main__":
    asyncio.run(main())
```

### Retry with Different Strategies

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def fetch_with_fallback(
        primary_source: str,
        backup_source: str,
        services=None
    ):
        """Try primary source, fallback to backup on failure."""
        try:
            # Try primary source
            return f"Data from {primary_source}"
        except Exception:
            # Fallback to backup
            return f"Data from {backup_source} (backup)"

    @app.workflow
    async def resilient_fetch(services=None):
        """Fetch with automatic fallback."""
        result = await fetch_with_fallback(
            primary_source="api.example.com",
            backup_source="cache",
            services=services
        )
        return {"data": result}

    await app.publish()

    result = await app.resilient_fetch().wait_result()
    print(result)

if __name__ == "__main__":
    asyncio.run(main())
```

## Error Aggregation

### Collect All Errors

When processing multiple items, collect all errors for review:

```python
import asyncio
from blazing import Blazing

async def main():
    app = Blazing()

    @app.step
    async def validate_item(item: dict, services=None):
        """Validate a single item."""
        errors = []

        if "id" not in item:
            errors.append("Missing required field: id")
        if "value" not in item or not isinstance(item["value"], int):
            errors.append("Invalid or missing value field")
        if item.get("value", 0) < 0:
            errors.append("Value must be non-negative")

        return {"item": item, "errors": errors, "valid": len(errors) == 0}

    @app.workflow
    async def batch_validation(items: list, services=None):
        """Validate multiple items and collect all errors."""
        results = []

        for item in items:
            validation = await validate_item(item, services=services)
            results.append(validation)

        valid_count = sum(1 for r in results if r["valid"])

        return {
            "total": len(items),
            "valid": valid_count,
            "invalid": len(items) - valid_count,
            "results": results
        }

    await app.publish()

    # Test with mixed data
    test_items = [
        {"id": 1, "value": 100},
        {"id": 2},  # Missing value
        {"value": -5},  # Missing id, negative value
    ]

    result = await app.batch_validation(test_items).wait_result()
    print(f"Valid: {result['valid']}/{result['total']}")

if __name__ == "__main__":
    asyncio.run(main())
```

## Best Practices

1. **Validate early** - Check inputs at workflow entry points
2. **Use specific exceptions** - Catch specific exception types, not bare `except`
3. **Provide context** - Include helpful error messages with details
4. **Implement timeouts** - Prevent workflows from hanging indefinitely
5. **Log errors** - Record failures for debugging and monitoring
6. **Graceful degradation** - Return partial results when possible
7. **Retry transient failures** - Use exponential backoff for network/service errors

## Common Exception Types

```python
# Input validation
ValueError        # Invalid values
TypeError         # Wrong types
KeyError          # Missing required data

# Resource access
ConnectionError   # Network failures
TimeoutError      # Operations too slow
FileNotFoundError # Missing files

# Logic errors
ZeroDivisionError # Division by zero
IndexError        # List access out of bounds
AttributeError    # Missing attributes
```
